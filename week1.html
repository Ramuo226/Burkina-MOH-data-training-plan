4ï¸âƒ£ Bonus Solution: Automated CSV Import Script

# Automated Malaria Data Analysis Script
# Load required libraries
library(readr)
library(writexl)

# Function to analyze malaria data
analyze_malaria_data <- function() {

# Ask user for file path
cat("=== Malaria Data Analysis Tool ===\n")
file_path <- readline(prompt = "Enter CSV file path: ")

# Check if file exists
if (!file.exists(file_path)) {
cat("Error: File not found!\n")
return(NULL)
}

# Load the dataset
cat("Loading dataset...\n")
tryCatch({
malaria_data <- read_csv(file_path)
cat("âœ“ Dataset loaded successfully!\n")
}, error = function(e) {
cat("Error loading file:", e$message, "\n")
return(NULL)
})

# Display dataset information
cat("\n=== Dataset Information ===\n")
cat("Rows:", nrow(malaria_data), "\n")
cat("Columns:", ncol(malaria_data), "\n")
cat("Column names:", paste(colnames(malaria_data), collapse = ", "), "\n")

# Display summary statistics
cat("\n=== Summary Statistics ===\n")
print(summary(malaria_data))

# Basic data cleaning
cat("\n=== Data Cleaning ===\n")
# Remove rows with missing values
original_rows <- nrow(malaria_data)
malaria_data <- na.omit(malaria_data)
cleaned_rows <- nrow(malaria_data)

cat("Removed", original_rows - cleaned_rows, "rows with missing values\n")
cat("Final dataset:", cleaned_rows, "rows\n")

# Save cleaned data
output_file <- paste0("cleaned_", basename(file_path))
write_csv(malaria_data, output_file)
cat("âœ“ Cleaned data saved as:", output_file, "\n")

# Create Excel report
report_data <- list(
"Cleaned_Data" = malaria_data,
"Summary" = data.frame(
Metric = c("Total Rows", "Total Columns", "Missing Values Removed"),
Value = c(cleaned_rows, ncol(malaria_data), original_rows - cleaned_rows)
)
)

excel_file <- paste0("malaria_report_", Sys.Date(), ".xlsx")
write_xlsx(report_data, excel_file)
cat("âœ“ Excel report saved as:", excel_file, "\n")

cat("\n=== Analysis Complete! ===\n")
return(malaria_data)
}

âœ” readline() prompts user for input

âœ” na.omit() removes rows with missing values
âœ” basename() extracts filename from full path
